import argparse
import datetime
import logging
import threading
import signal
import sys
import os
from collections import defaultdict
from scapy.all import sniff, IP, TCP, UDP, Raw

# ====================== CONFIG ======================
logging.basicConfig(
    filename='netlog.db',
    level=logging.INFO,
    format='%(message)s'  # Minimal format for stealth
)

stats = defaultdict(int)
lock = threading.Lock()
running = True

KEYWORDS = ['password', 'passwd', 'login', 'token', 'auth', 'api_key', 'secret', 'private']

def packet_handler(packet):
    if not running or IP not in packet:
        return
    
    src = packet[IP].src
    dst = packet[IP].dst
    
    with lock:
        stats['total'] += 1
        if TCP in packet:
            stats['tcp'] += 1
            port = packet[TCP].dport
        elif UDP in packet:
            stats['udp'] += 1
            port = packet[UDP].dport
        else:
            return

    # Very light processing
    if Raw in packet:
        try:
            payload = packet[Raw].load.lower()
            for kw in KEYWORDS:
                if kw in payload:
                    logging.info(f"SENSITIVE|{datetime.datetime.now().strftime('%H:%M:%S')}|{src}->{dst}|{kw}")
                    break
        except:
            pass

    # Optional: very quiet per-packet logging (comment out for max stealth)
    # logging.info(f"{src}->{dst}|{port}")

def stats_reporter():
    while running:
        threading.Event().wait(20)
        with lock:
            if stats['total'] > 0:
                logging.info(f"STATS|Total:{stats['total']}|TCP:{stats['tcp']}|UDP:{stats['udp']}")

def main():
    parser = argparse.ArgumentParser(add_help=False)  # Less obvious help
    parser.add_argument("-i", "--iface")
    parser.add_argument("-f", "--filter", default="")
    parser.add_argument("-o", "--output", default="capture.pcap")
    parser.add_argument("-t", "--time", type=int, default=0)
    parser.add_argument("--daemon", action="store_true", help="Run silently")
    args = parser.parse_args()

    if args.daemon:
        logging.getLogger().setLevel(logging.WARNING)  # Almost nothing on console

    def stop(sig, frame):
        global running
        running = False
        print("\n[+] Shutting down cleanly...")
        sys.exit(0)

    signal.signal(signal.SIGINT, stop)
    signal.signal(signal.SIGTERM, stop)

    # Background threads
    threading.Thread(target=stats_reporter, daemon=True).start()

    try:
        sniff(
            iface=args.iface,
            filter=args.filter,
            prn=packet_handler,
            store=False,
            timeout=args.time if args.time > 0 else None,
            stop_filter=lambda x: not running
        )
    except Exception as e:
        logging.error(f"ERR|{e}")

if __name__ == "__main__":
    main()
